\documentclass[convert]{standalone}

\usepackage{tikz}
\pagestyle{empty}

\begin{document}
\begin{tikzpicture}[> = latex, font = \scriptsize]

	% Definitions
	
	\def\xi{-2.5}		% Initial x, y coordinates
	\def\yi{2.5}
	
	\def\vx{1.5}		% Velocity components
	\def\vy{-0.5}
	
	% Draw ball
	
	\foreach \n in {0, 1, ..., 4}
	{
		\draw [ball color = green!50, draw = none] ({\xi + \n * \vx}, {\yi + \n * \vy}) circle (4 pt) node [above right] {$t_\n$};
		\draw [->, red] (0, 0) --  ({\xi + \n * \vx}, {\yi + \n * \vy});
	}
	
	\foreach \n in {0, 1, ..., 3}
		\draw [->, blue] ({\xi + \n * \vx}, {\yi + \n * \vy}) -- node [above] {${\vec v} \Delta t$} ({\xi + (\n + 1) * \vx}, {\yi + (\n + 1) * \vy});;
	
	% Draw origin
	
	\filldraw (0, 0) circle (1 pt) node [below left] {$O$};
	
\end{tikzpicture}
\end{document}
