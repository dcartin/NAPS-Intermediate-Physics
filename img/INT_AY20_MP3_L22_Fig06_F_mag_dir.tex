	\def\delta{0.125}		% Size of x's for magnetic field
	
	\foreach \x in {-1.875, -1.125, ..., 1.875}
	\foreach \y in {-1, 0, 1}
		{
		\begin{scope}[very thick, blue]
		
			\draw (\x + \delta, \y + \delta) -- (\x - \delta, \y - \delta);
			\draw (\x + \delta, \y - \delta) -- (\x - \delta, \y + \delta);
			
		\end{scope}
		}
		
	\node at (1.875, 1) [right = 0.5 em, blue] {${\vec B}$};
